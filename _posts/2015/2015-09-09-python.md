## 解析html，解析table表格

{% highlight python %}
from bs4 import BeautifulSoup

soup = BeautifulSoup(open(html_fname))
tbodys = soup.findChildren('tbody')
tbody = tbodys[0]

rows = tbody.findChildren('tr')
tb = []
for row in rows:
     r = []
     cells = row.findChildren('td')
     for cell in cells:
             value = cell.string.strip()
             r.append(value)
     
     tb.append(r)
#tb is table cell dim2 array
{% endhighlight %}
