{% highlight perl %}
#!/usr/bin/perl
use Coro;

my $arr = [ 1, 3, 5 , 4 , 2 ,7 ];
my @tasks;
my @res;
for my $i (0 .. $#$arr){
    push @tasks, async {
        my $x = $arr->[$i];
        my $s = $x*$x;
        $res[$i] = $s;
    }
}

for (@tasks){
    $_->join;
}

print $_, "\n" for @res;
